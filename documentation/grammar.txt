### set of productions
  stmt      ->  T_ID = expr T_SEMICOLON
              | T_IF T_LEFT_PAREN expr T_RIGHT_PAREN stmt
              | T_IF T_LEFT_PAREN expr T_RIGHT_PAREN stmt T_ELSE stmt
              | T_WHILE T_LEFT_PAREN expr T_RIGHT_PAREN stmt
              | T_LEFT_BRACE stmts T_RIGHT_BRACE
              | { stmt }
  stmts     ->  stmts stmt
              | e
  call      ->  T_ID T_LEFT_PAREN optparams T_RIGHT_PAREN
  optparams ->  params | e
  params    ->  params, param | param

  expr      ->  expr T_PLUS term | expr T_MINUS term | term
  term      ->  term T_STAR factor | term T_SLASH factor | factor
  factor    ->  T_NUMBER | T_LEFT_PAREN expr T_RIGHT_PAREN
